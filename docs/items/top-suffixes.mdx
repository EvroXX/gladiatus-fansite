import ItemAffixList from '@site/src/components/ItemAffixList';
import suffixes from '@site/static/data/items/suffixes.json';

export const topSuffixNames = [
  'of Gloom',
  'of the Inferno',
  'of the Dragon',
  'of Aggression',
  'of Elimination',
  'of Earth',
  'of Delicacy',
];

export const topSuffixes = topSuffixNames
  .map((name) => suffixes.find((suffix) => suffix.name === name))
  .filter(Boolean);

export const getPercent = (item, stat) => item?.stats?.[stat]?.percent ?? 0;
export const getFlat = (item, stat) => item?.stats?.[stat]?.flat ?? 0;

export const topByPercent = (items, stat, minPercent) =>
  items
    .filter((item) => getPercent(item, stat) >= minPercent)
    .sort((a, b) => getPercent(b, stat) - getPercent(a, stat) || (b.level ?? 0) - (a.level ?? 0))
    .slice(0, 5);

export const topIntelligenceOrHealing = suffixes
  .filter((item) => getPercent(item, 'intelligence') >= 30 || getFlat(item, 'healing') > 0 || getPercent(item, 'healing') > 0)
  .sort(
    (a, b) =>
      getPercent(b, 'intelligence') - getPercent(a, 'intelligence') ||
      (getFlat(b, 'healing') + getPercent(b, 'healing')) - (getFlat(a, 'healing') + getPercent(a, 'healing')) ||
      (b.level ?? 0) - (a.level ?? 0)
  )
  .slice(0, 5);

export const topDamageCriticalCombos = suffixes
  .filter(
    (item) =>
      getFlat(item, 'damage') > 0 &&
      (getFlat(item, 'critical_attack_value') > 0 ||
        getPercent(item, 'critical_attack_value') > 0 ||
        getFlat(item, 'critical_healing_value') > 0 ||
        getPercent(item, 'critical_healing_value') > 0)
  )
  .sort(
    (a, b) =>
      getFlat(b, 'damage') - getFlat(a, 'damage') ||
      (getFlat(b, 'critical_attack_value') +
        getPercent(b, 'critical_attack_value') +
        getFlat(b, 'critical_healing_value') +
        getPercent(b, 'critical_healing_value')) -
        (getFlat(a, 'critical_attack_value') +
          getPercent(a, 'critical_attack_value') +
          getFlat(a, 'critical_healing_value') +
          getPercent(a, 'critical_healing_value')) ||
      (b.level ?? 0) - (a.level ?? 0)
  )
  .slice(0, 5);
